type CompanyGraph {
  employees(id: ID, ids: [ID], orderBy: [OrderBy], where: [WhereExpression], skip: Int, take: Int): [EmployeeGraph!]!
  employeesConnection(after: String, first: Int, before: String, last: Int, where: [WhereExpression], orderBy: [OrderBy], ids: [ID]): EmployeeGraphConnection
  content: String
  id: Int!
}

type CompanyGraphConnection {
  totalCount: Int
  pageInfo: PageInfo!
  edges: [CompanyGraphEdge]
  items: [CompanyGraph]
}

type CompanyGraphEdge {
  cursor: String!
  node: CompanyGraph
}

enum Comparison {
  contains
  endsWith
  equal
  greaterThan
  greaterThanOrEqual
  notIn
  in
  lessThan
  lessThanOrEqual
  like
  startsWith
}

enum Connector {
  and
  or
}

type EmployeeGraph {
  company: CompanyGraph!
  age: Int!
  companyId: Int!
  content: String
  id: Int!
}

type EmployeeGraphConnection {
  totalCount: Int
  pageInfo: PageInfo!
  edges: [EmployeeGraphEdge]
  items: [EmployeeGraph]
}

type EmployeeGraphEdge {
  cursor: String!
  node: EmployeeGraph
}

type EmployeeSummaryGraph {
  averageAge: Float!
  companyId: Int!
}

input OrderBy {
  path: String!
  descending: Boolean
}

type PageInfo {
  hasNextPage: Boolean!
  hasPreviousPage: Boolean!
  startCursor: String
  endCursor: String
}

type Query {
  companies(id: ID, ids: [ID], orderBy: [OrderBy], where: [WhereExpression], skip: Int, take: Int): [CompanyGraph!]!
  company(id: ID, ids: [ID], orderBy: [OrderBy], where: [WhereExpression], skip: Int, take: Int): CompanyGraph!
  companyOrNull(id: ID, ids: [ID], orderBy: [OrderBy], where: [WhereExpression], skip: Int, take: Int): CompanyGraph
  companiesConnection(after: String, first: Int, before: String, last: Int, where: [WhereExpression], orderBy: [OrderBy], ids: [ID]): CompanyGraphConnection
  employees(id: ID, ids: [ID], orderBy: [OrderBy], where: [WhereExpression], skip: Int, take: Int): [EmployeeGraph!]!
  employeesByArgument(id: ID, ids: [ID], orderBy: [OrderBy], where: [WhereExpression], skip: Int, take: Int, content: String): [EmployeeGraph!]!
  employeesConnection(after: String, first: Int, before: String, last: Int, where: [WhereExpression], orderBy: [OrderBy], ids: [ID]): EmployeeGraphConnection
  employeeSummary(where: [WhereExpression]): [EmployeeSummaryGraph]
}

enum StringComparison {
  CURRENT_CULTURE
  CURRENT_CULTURE_IGNORE_CASE
  INVARIANT_CULTURE
  INVARIANT_CULTURE_IGNORE_CASE
  ORDINAL
  ORDINAL_IGNORE_CASE
}

input WhereExpression {
  path: String
  comparison: Comparison
  negate: Boolean
  case: StringComparison
  value: [String]
  connector: Connector
  groupedExpressions: [WhereExpression]
}
